import numpy as np
from sklearn.model_selection import train_test_split

n_class, n_len, width, height = 11, 5, 28, 28

def generate_dataset(X, y):
    X_len = X.shape[0]
    
    X_gen = np.zeros((X_len, height, width*n_len, 1), dtype=np.uint8)
    y_gen = [np.zeros((X_len, n_class), dtype=np.uint8) for i in range(n_len)]
    # TODO: 1~5
    for i in range(X_len):
        n = random.randint(1, n_len)  # 
        for j in range(n):
            index = random.randint(0, X_len-1)
            X_gen[i, :, width*j:width*(j+1), 0] = X[index]
            y_gen[j][i, y[index]] = 1
        for j in range(n, n_len):
            y_gen[j][i, 10] = 1  # 10    
    return X_gen, y_gen

X_raw_train, X_raw_valid, y_raw_train, y_raw_valid = train_test_split(X_raw, y_raw, test_size=0.2, random_state=42)

X_train, y_train = generate_dataset(X_raw_train, y_raw_train)
X_valid, y_valid = generate_dataset(X_raw_valid, y_raw_valid)
X_test, y_test = generate_dataset(X_raw_test, y_raw_test)

inputs = Input(shape=(height, width*n_len, 1))

x = Conv2D(32, 3, strides=1, padding='same', activation='relu', kernel_initializer='truncated_normal', bias_initializer='truncated_normal')(x)
x = MaxPooling2D(2, 2, 'same')(x)
x = Dropout(0.25)(x)

x = Conv2D(64, 3, strides=1, padding='same', activation='relu', kernel_initializer='truncated_normal', bias_initializer='truncated_normal')(x)

x = MaxPooling2D(2, 2, 'same')(x)
x = Dropout(0.25)(x)

x = Flatten()(x)
x = Dense(256, activation='relu', kernel_initializer='truncated_normal', bias_initializer='truncated_normal')(x)
x = Dropout(0.5)(x)

def evaluate(model, X, y):
    # TODO: 
    y_pred = model.predict(X)
    compare = np.argmax(y_pred, axis=2) == np.argmax(y, axis=2)
    compare = compare.all(axis=0)
    count = np.count_nonzero(compare)
    return float(count) / len(compare)
